			this.play(this.prev,this.current);
		},
		playNext:function(){
			if(!this.contNode[0].checked){
				return;
			}
			this.prev=this.current;
			if(this.current>=this.length-1){
				return;
			}
			this.current++;
			this.play(this.prev,this.current);
		},
		play:function(prev,current){
			this.player.playVideo(this.videoList[current]);
			this.titleNode.html(this.videoList[current].title);
			this.playListNode.eq(prev).removeClass("selected");
			this.playListNode.eq(current).addClass("selected");
		}
	}

　　开车时iPhone两次爆炸车载记录仪全程拍下

　　【环球网科技记者 张之颖】日前，一位上海江姓女士开车行驶的过程中，放在仪表台上的iPhone突然爆炸起火。这则消息今日引起外媒的关注。　
　　据外媒Thrillist报道，事发当时江女士正和丈夫一同开车，即使车中开着空调，当时室外气温达到37摄氏度，不知道是否因此影响了电池的温度，iPhone突然爆炸起火。过了十几秒后这台iPhone又二次起火，由于乘客受到惊吓，迫使该乘客将车停靠至路边，车载记录仪全程记下过程。
　　事后江女士接受采访时表示，2月份从一家未经认证的维修店更换了iPhone6电池。iPhone的电池并非来自于苹果官方生产，该电池被发现是导致爆炸的原因。
　　第三方维修人员拒绝对此事负责，并声称电池为真，不存在质量问题。尽管如此，由于这不是非官方更换电池的iPhone第一次起火，苹果自然也不会承担任何责任。外媒提醒，你应该确保将iPhone带到苹果授权的维修店。　
　　同时，外媒也戏谑地称，即使是官方iPhone6S也有爆炸的历史纪录，或许人们要将手机丢到河里，干脆在树林中生活，才不会遇到类似的事情。
