			var_this=this;
			$("#J_Play_List").on("click",".J_Play_Itema",function(e){
				e.preventDefault();
				_this.playCurrent($(this));
			});
		},
		initPlayer:function(){
			var_this=this;
			this.player=this.playVideo(this.flashConfig);
			this.player.init(this.videoList[this.prev]);
			this.player.on("playCompleted",function(){
				_this.playNext();
			});
			this.playListNode.eq(0).addClass("selected");
			this.titleNode.html(this.videoList[0].title);
		},
		playCurrent:function($this){
			this.prev=this.current;
			this.current=$this.parents(".J_Play_Item").index();
			this.play(this.prev,this.current);
		},
		playNext:function(){
			if(!this.contNode[0].checked){
				return;
			}
			this.prev=this.current;
			if(this.current>=this.length-1){
				return;
			}
			this.current++;
			this.play(this.prev,this.current);
		},
		play:function(prev,current){
			this.player.playVideo(this.videoList[current]);
			this.titleNode.html(this.videoList[current].title);
			this.playListNode.eq(prev).removeClass("selected");
			this.playListNode.eq(current).addClass("selected");
		}
	}
	Play.init(flashConfig,videoList);
})(jQuery);




　　新一批新车安全碰撞评价结果公布安全性参差不齐
　　央视网9月10日消息：中国汽车技术研究中心9号公布了今年第三批新车安全碰撞评价结果。结果显示，有部分自主品牌车型成绩不够理想。
　　这批安全碰撞测试共有6款新车型，其中3款为合资品牌车型，3款为自主品牌车型。测试结果显示：3个五星级，2个四星级，1个三星级，安全性总体表现参差不齐，有些自主品牌车型过于强调成本控制，成绩不够理想。
　　中国汽车技术研究中心书记于凯：我们一贯提倡，应该把安全设施投入，要大力加强，不能以减低安全配置来降低成本，这是不可取的。
　　中国汽车技术研究中心同时表示，这批次的安全碰撞测试的车型选取的都是在售、销量较大的新车型，能够反映出该款车型实际安全水平。在下一批测试车型中，计划增加进口车型、合资高端车型的测试，使测试车型覆盖面更广。
